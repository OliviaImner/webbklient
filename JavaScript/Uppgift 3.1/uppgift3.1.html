RegExp

Text kan bytas ut igenom att använda replace, i syntaxen nedanför visas hur "sämst" byts it till "bäst" genom att trycka på knappen visa.


function regEx() {
    var str = document.getElementById("reg").innerHTML; 
    var txt = str.replace(/sämst/i, "bäst");
    document.getElementById("reg").innerHTML = txt;
}
